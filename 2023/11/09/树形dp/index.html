树形dp

换根dp

树形 DP 中的换根 DP 问题又被称为二次扫描，通常不会指定根结点，并且根结点的变化会对一些值，例如子结点深度和、点权和等产生影响。

通常需要两次 DFS，第一次 DFS 预处理诸如深度，点权和之类的信息，在第二次 DFS 开始运行换根动态规划。

834. 树中距离之和

class Solution {
    int[] res;
    int[] size;
    public void dfs(int cur,int fa,List<Integer>[] g,int depth){
        res[0] += depth;
        size[cur] = 1;
        for(int ch:g[cur]){
            if(ch == fa)
                continue;
            dfs(ch,cur,g,depth+1);
            size[cur] += size[ch];
        }

    }
    public void dfs2(int cur, int fa,List<Integer>[] g){
        for(int ch:g[cur]){
            if(ch != fa){
                res[ch] = res[cur]-2*size[ch]+g.length;
                dfs2(ch,cur,g);
            }
        }
    }
    public int[] sumOfDistancesInTree(int n, int[][] edges) {
        res = new int[n];
        size = new int[n];
        List<Integer>[] g = new ArrayList[n];
        Arrays.setAll(g,e->new ArrayList<>());
        for(int[] edge:edges){
            g[edge[0]].add(edge[1]);
            g[edge[1]].add(edge[0]);
        }
        dfs(0,-1,g,0);
        dfs2(0,-1,g);
        return res;
    }
}
